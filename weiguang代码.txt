<script>
const SUPABASE_URL = "https://kikzvlvlvuafejfzcjbc.supabase.co";
const SUPABASE_KEY = "sb_publishable_WsQdzSnaI_D-H8QNt8qX0w_X76iyjar";
const STORAGE_KEYS = {
  userId: "emotion_mvp_user_id",
  history: "emotion_mvp_history",
  feedback: "emotion_mvp_feedback"
};

const els = {
 
  userInput: document.getElementById("userInput"),
  submitBtn: document.getElementById("submitBtn"),
  homeStatus: document.getElementById("homeStatus"),
  conversationThread: document.getElementById("conversationThread"),
  originalText: document.getElementById("originalText"),
  followupInput: document.getElementById("followupInput"),
  continueChatBtn: document.getElementById("continueChatBtn"),
  endChatBtn: document.getElementById("endChatBtn"),
  helpfulBtn: document.getElementById("helpfulBtn"),
  notHelpfulBtn: document.getElementById("notHelpfulBtn"),
  feedbackActions: document.getElementById("feedbackActions"),
  feedbackList: document.getElementById("feedbackList"),
  historyList: document.getElementById("historyList"),
  resultStatus: document.getElementById("resultStatus"),
  backHomeFromFeedbackBtn: document.getElementById("backHomeFromFeedbackBtn")
};

let session = null;
let historyRecords = [];
let feedbackRecords = [];
let supabaseClient = null;

function getUserId() {
  let id = localStorage.getItem(STORAGE_KEYS.userId);
  if (!id) {
    id = "user_" + Date.now() + "_" + Math.random().toString(36).slice(2, 10);
    localStorage.setItem(STORAGE_KEYS.userId, id);
  }
  return id;
}

function showPage(id) {
  document.querySelectorAll(".page").forEach(p => p.classList.remove("active"));
  document.getElementById(id).classList.add("active");
}

function saveLocal() {
  localStorage.setItem(STORAGE_KEYS.history, JSON.stringify(historyRecords));
  localStorage.setItem(STORAGE_KEYS.feedback, JSON.stringify(feedbackRecords));
}

function loadLocal() {
  getUserId();
  try {
    historyRecords = JSON.parse(localStorage.getItem(STORAGE_KEYS.history) || "[]") || [];
  } catch {
    historyRecords = [];
  }
  try {
    feedbackRecords = JSON.parse(localStorage.getItem(STORAGE_KEYS.feedback) || "[]") || [];
  } catch {
    feedbackRecords = [];
  }
  initSupabase();
}

function initSupabase() {
  if (!SUPABASE_URL || !SUPABASE_KEY) {
    supabaseClient = null;
    console.error("Supabase 配置缺失");
    return;
  }

  if (!window.supabase || !window.supabase.createClient) {
    supabaseClient = null;
    console.error("Supabase 脚本加载失败");
    return;
  }

  try {
    supabaseClient = window.supabase.createClient(SUPABASE_URL, SUPABASE_KEY);
    console.log("Supabase 已连接");
  } catch (e) {
    supabaseClient = null;
    console.error("Supabase 初始化失败：", e.message || e);
  }
}
 

async function upsertConversationToDB(record) {
  if (!supabaseClient) {
    console.error("Supabase 未连接");
    return false;
  }

  const payload = {
    user_id: getUserId(),
    conversation_id: String(record.id),
    first_user_text: record.messages?.[0]?.content || "",
    messages: null,
    ended: !!record.ended,
    feedback: record.feedback || null,
    message_count: record.messages?.length || 0,
    updated_at: new Date().toISOString()
  };

  const { error } = await supabaseClient
    .from("conversations")
    .upsert(payload, { onConflict: "conversation_id" });

  if (error) {
    console.error("保存 conversations 失败：", error.message);
    return false;
  }

  return true;
}
async function insertFeedbackToDB(record, type) {
  if (!supabaseClient) {
    console.error("Supabase 未连接");
    return false;
  }

  const payload = {
    user_id: getUserId(),
    conversation_id: String(record.id),
    feedback_type: type,
    first_user_text: record.messages?.[0]?.content || "",
    messages: record.messages || [],
    message_count: record.messages?.length || 0
  };

  const { error } = await supabaseClient
    .from("feedback_records")
    .insert(payload);

  if (error) {
    console.error("保存 feedback_records 失败：", error.message);
    return false;
  }

  return true;
}
function escapeHtml(str = "") {
  return str.replace(/&/g, "&amp;").replace(/</g, "&lt;").replace(/>/g, "&gt;");
}

function renderThread() {
  if (!session || !session.messages) {
    els.conversationThread.innerHTML = '<div class="empty">还没有对话内容</div>';
    return;
  }
  els.conversationThread.innerHTML = session.messages.map(m => `
    <div class="bubble ${m.role}">
      <strong>${m.role === "user" ? "你" : "AI"}</strong>
      <div style="margin-top:6px">${escapeHtml(m.content)}</div>
    </div>
  `).join("");
}

function renderHistory() {
  els.historyList.innerHTML = historyRecords.length
    ? historyRecords.slice().reverse().map(item => `
      <div class="panel">
        <div><strong>匿名用户 ID：</strong>${escapeHtml(item.user_id || "")}</div>
        <div class="meta"><strong>首条倾诉：</strong>${escapeHtml(item.messages?.[0]?.content || "")}</div>
        <div class="meta"><strong>对话轮数：</strong>${item.messages?.length || 0}</div>
        <div class="meta"><strong>状态：</strong>${item.ended ? "已结束" : "进行中"}</div>
      </div>
    `).join("")
    : '<div class="empty">暂无历史记录</div>';
}

function renderFeedback() {
  els.feedbackList.innerHTML = feedbackRecords.length
    ? feedbackRecords.slice().reverse().map(item => `
      <div class="panel">
        <div><strong>反馈：</strong>${item.type === "good" ? "有帮助" : "没帮助"}</div>
        <div class="meta"><strong>匿名用户 ID：</strong>${escapeHtml(item.user_id || "")}</div>
        <div class="meta"><strong>首条倾诉：</strong>${escapeHtml(item.messages?.[0]?.content || "")}</div>
        <div class="meta"><strong>对话轮数：</strong>${item.messages?.length || 0}</div>
      </div>
    `).join("")
    : '<div class="empty">暂无反馈记录</div>';
}

async function send(msgs) {
  const res = await fetch("/api/chat", {
    method: "POST",
    headers: {
      "Content-Type": "application/json"
    },
    body: JSON.stringify({
      messages: msgs
    })
  });

  const data = await res.json();
  if (!res.ok) {
    throw new Error(data?.error?.message || data?.message || "请求失败");
  }
  return data?.reply || "emm…我刚刚有点卡壳了，你再和我说一次，我这次认真接住你。";
}



els.submitBtn.onclick = async () => {
  const text = els.userInput.value.trim().slice(0, 300);
  if (!text) {
    els.homeStatus.textContent = "先输入一点内容呀";
    return;
  }

  try {
    els.homeStatus.textContent = "";
    session = {
      id: Date.now(),
      user_id: getUserId(),
      messages: [{ role: "user", content: text }],
      ended: false,
      feedback: null
    };
    els.originalText.textContent = text;
    els.feedbackActions.style.display = "none";
    els.followupInput.style.display = "block";
    els.followupInput.value = "";
    els.resultStatus.textContent = "AI 正在回复你…";
    showPage("resultPage");

    const reply = await send(session.messages);
    session.messages.push({ role: "assistant", content: reply });
    renderThread();
    els.resultStatus.textContent = "你可以继续聊下去，想结束时再点“结束对话”。";

    const idx = historyRecords.findIndex(item => item.id === session.id);
    if (idx === -1) historyRecords.push(JSON.parse(JSON.stringify(session)));
    else historyRecords[idx] = JSON.parse(JSON.stringify(session));

    saveLocal();
    renderHistory();
    await upsertConversationToDB(session);
  } catch (err) {
    showPage("homePage");
    els.homeStatus.textContent = err.message || "出错了";
  }
};

els.continueChatBtn.onclick = async () => {
  if (!session || session.ended) {
    els.resultStatus.textContent = "这轮对话已经结束啦，想继续的话请从首页开启新对话。";
    return;
  }
  const text = els.followupInput.value.trim().slice(0, 300);
  if (!text) {
    els.resultStatus.textContent = "你还没输入继续聊的内容哦";
    return;
  }

  try {
    session.messages.push({
 role:"user",
 content:text.slice(0,200)
});
    els.followupInput.value = "";
    renderThread();
    els.resultStatus.textContent = "AI 正在继续接话…";

    const reply = await send(session.messages.slice(-4));
    session.messages.push({ role: "assistant", content: reply });
    renderThread();
    els.resultStatus.textContent = "还可以继续聊，次数不限制。";

    const idx = historyRecords.findIndex(item => item.id === session.id);
    if (idx === -1) historyRecords.push(JSON.parse(JSON.stringify(session)));
    else historyRecords[idx] = JSON.parse(JSON.stringify(session));

    saveLocal();
    renderHistory();
    await upsertConversationToDB(session);
  } catch (err) {
    els.resultStatus.textContent = err.message || "继续对话失败";
  }
};

els.endChatBtn.onclick = async () => {
  if (!session) {
    els.resultStatus.textContent = "当前还没有可结束的对话";
    return;
  }
  session.ended = true;
  els.feedbackActions.style.display = "flex";
  els.followupInput.style.display = "none";
  els.resultStatus.textContent = "对话已结束，请选择这次回复对你是否有帮助。";

  const idx = historyRecords.findIndex(item => item.id === session.id);
  if (idx === -1) historyRecords.push(JSON.parse(JSON.stringify(session)));
  else historyRecords[idx] = JSON.parse(JSON.stringify(session));

  saveLocal();
  renderHistory();
  await upsertConversationToDB(session);
};

els.helpfulBtn.onclick = async () => {
  if (!session || !session.ended) {
    els.resultStatus.textContent = "请先结束对话，再提交反馈。";
    return;
  }
  feedbackRecords.push({ ...JSON.parse(JSON.stringify(session)), type: "good", user_id: getUserId() });
  session.feedback = "good";
  saveLocal();
  renderFeedback();
  await upsertConversationToDB(session);
  await insertFeedbackToDB(session, "good");
  showPage("feedbackPage");
};

els.notHelpfulBtn.onclick = async () => {
  if (!session || !session.ended) {
    els.resultStatus.textContent = "请先结束对话，再提交反馈。";
    return;
  }
  feedbackRecords.push({ ...JSON.parse(JSON.stringify(session)), type: "bad", user_id: getUserId() });
  session.feedback = "bad";
  saveLocal();
  renderFeedback();
  await upsertConversationToDB(session);
  await insertFeedbackToDB(session, "bad");
  showPage("feedbackPage");
};

els.backHomeFromFeedbackBtn.onclick = () => {
  showPage("homePage");
};

document.querySelectorAll("[data-page]").forEach(btn => {
  btn.onclick = () => showPage(btn.dataset.page);
});

window.addEventListener("error", (e) => {
  console.error("页面报错：", e.message);
});

loadLocal();
renderHistory();
renderFeedback();
</script>
